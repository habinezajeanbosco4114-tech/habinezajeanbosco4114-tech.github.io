Advanced C++ Programming (OOP)

1. Object Oriented Programming (OOP)

OOP is a programming style based on objects and classes.

2. Class and Object

#include 
using namespace std;

class Student {
public:
    string name;
    int age;

    void display() {
        cout << "Name: " << name << endl;
        cout << "Age: " << age << endl;
    }
};

int main() {
    Student s1;
    s1.name = "John";
    s1.age = 20;
    s1.display();

    return 0;
}

3. Constructor

class Car {
public:
    string brand;

    Car(string b) {
        brand = b;
    }
};

int main() {
    Car c1("Toyota");
    cout << c1.brand;
}

4. Encapsulation

class Person {
private:
    int age;

public:
    void setAge(int a) {
        age = a;
    }

    int getAge() {
        return age;
    }
};

5. Inheritance

class Animal {
public:
    void sound() {
        cout << "Animal sound";
    }
};

class Dog : public Animal {
};

int main() {
    Dog d;
    d.sound();
}

6. Polymorphism

class Shape {
public:
    virtual void draw() {
        cout << "Drawing shape";
    }
};

class Circle : public Shape {
public:
    void draw() {
        cout << "Drawing circle";
    }
};
