1. Class and Object

    void display() {
        cout << name << age;
    }
};

2. Constructor

class Car {
public:
    string brand;

    Car(string b) {
        brand = b;
    }
};

3. Encapsulation

class Person {
private:
    int age;

public:
    void setAge(int a) {
        age = a;
    }

    int getAge() {
        return age;
    }
};

4. Inheritance

class Animal {
public:
    void sound() {
        cout << "Animal sound";
    }
};

class Dog : public Animal {
};

5. Polymorphism

class Shape {
public:
    virtual void draw() {
        cout << "Drawing shape";
    }
